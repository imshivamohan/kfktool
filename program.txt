    except KafkaError as e:
        if e.code() == KafkaError.VALUE_SERIALIZATION:
            print("Kafka value serialization error:")
            print(f"Error code: {e.code()}")
            print(f"Error value: {e.value()}")
            print(f"Error string: {e.str()}")
        else:
            print(f"Kafka error: {e}")
